def send_tron_usdt(to_address: str, amount: float):
    # Load hot wallet key
    priv_key = PrivateKey(bytes.fromhex(HOT_WALLET_KEY))
    
    # USDT contract interaction
    contract = tron.get_contract(usdt_contract)
    
    # Convert amount to smallest unit (6 decimals for USDT)
    amount_sun = int(amount * 1_000_000)
    
    # Build transaction
    txn = (
        contract.functions.transfer(to_address, amount_sun)
        .with_owner(HOT_WALLET_ADDRESS)
        .fee_limit(50_000_000)  # 50 TRX max fee
        .build()
        .sign(priv_key)
    )
    
    # Broadcast
    result = txn.broadcast()
    
    return result.txid
